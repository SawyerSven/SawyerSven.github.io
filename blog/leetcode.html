// 双for粗暴解法
var numJewelsInStones = function (J, S) {
    let num = 0;

        }

}

# #709 转换小写字母

题目见转换小写字母

/**
 * 实现ToLowerCase()
 * 接收字符串参数str,并将字符串中的大写字母转换成小写字母
 * 返回新字符串
 */

// 神经病写法
var toLowerCase = function (str) {
    let map = {
        A: 'a', B: 'b', C: 'c', D: 'd', E: 'e', F: 'f', G: 'g', H: 'h',
        I: 'i', J: 'j', K: 'k', L: 'l', M: 'm', N: 'n', O: 'o', P: 'p', Q: 'q', R: 'r', S: 's',
        T: 't', U: 'u', V: 'v', W: 'w', X: 'x', Y: 'y', Z: 'z'
    }
    return [...str].map((val) => {
        return val = map[val] ? map[val] : val
    }).join('')

}

/**
 * A-Z:65-90
 * a-z:97-122
 */

var toLowerCase = function (str) {
    return [...str].map((val) => {
        if (val.charCodeAt() > 64 && val.charCodeAt() < 91) {
            return String.fromCharCode(val.charCodeAt() + 32)
        } else {
            return val
        }
    }).join('')
}

# #977 有序数组的平方

原题有序数组的平方

/**
 * 非递减顺序排序的整数数组A
 * 返回平方组成新数组，
 * 要求按非递减顺序排序
 */

 // map方法
var sortedSquares = function (A) {
    return A.map(v => v * v).sort((a, b) => a - b)
}

// a-z:97-122

var uniqueMorseRepresentations = function (words) {
    let dict = [".-", "-...", "-.-.", "-..", ".", "..-.", "--.", "....", "..", ".---", "-.-", ".-..", "--", "-.", "---", ".--.", "--.-", ".-.", "...", "-", "..-", "...-", ".--", "-..-", "-.--", "--.."];
    let result = new Set();
    for (let word of words) {
        let arr = [];
        for (let letter of word) {
            arr.push(dict[letter.charCodeAt() - 97])
        }
        arr = arr.join('')
        result.add(arr);
    }
    return result.size;

};
console.log(uniqueMorseRepresentations(["gin", "zen", "gig", "msg"]));

# #905 按奇偶排序数组

原题按奇偶排序数组

/**
 * 给定一个非负整数数组 A，返回一个由 A 的所有偶数元素组成的数组，后面跟 A 的所有奇数元素。
 *
 *你可以返回满足此条件的任何数组作为答案。
 */

/**
* @param {number[]} A
* @return {number[]}
*/
var sortArrayByParity = function (A) {
    let a = [];
    A.forEach(v => v % 2 === 0 ? a.unshift(v) : a.push(v));
    return a
};

 * * 解释: 你可以通过把第一个4变成1来使得它成为一个非递减数列。
 * * 示例 2:
 * * 输入: [4,2,1]
 * * 输出: False
 * * 解释: 你不能在只改变一个元素的情况下将其变为非递减数列。
 */

/**
 * @param {number[]} nums
 * @return {boolean}
 */
var checkPossibility = function (nums) {
    if (nums.length < 3) return true;
    let changeCount = 0;
    if (nums[0] > nums[1]) {
        nums[0] = nums[1]
        changeCount++;
    }
    for (let i = 0; i < nums.length; i++) {
        if (nums[i] > nums[i + 1]) {
            changeCount++;
            if (changeCount > 1) return false;
            let left = nums[i - 1];
            if (left > nums[i + 1]) {
                nums[i + 1] = nums[i];
            } else {
                nums[i] = left
            }
        }
    }
    return true
};
